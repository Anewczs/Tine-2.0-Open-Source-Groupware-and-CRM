  78.     /**
  79.      * get the grants for the user identified by $accountId for a specific application
  80.      *
  81.      * @param int $accountId the accountid of the user
  82.      * @param string $appName the name of the application to return the rights for
  83.      * @param bool $enumerateGroupAcls if TRUE the acl for the groupmembers gets returned too
  84.      * @return array the grants
  85.      */
  86.     public function getApplicationGrants($accountId, $appName, $enumerateGroupAcls = TRUE)
  87.     {
  88.         $accounts = new Egwbase_Account_Sql();
  89.         $groupMemberships = $accounts->getAccountGroupMemberships($accountId);
  90.         $groupMemberships[] = $accountId;
  91.         
  92.         $aclTable = new Egwbase_Acl_Sql();
  93.         $where = array(
  94.             $aclTable->getAdapter()->quoteInto('acl_appname = ?', $appName),
  95.             // who gave rights to me and my groups
  96.             $aclTable->getAdapter()->quoteInto('acl_location IN (?)', $groupMemberships)
  97.         );
  98.         $rowSet = $aclTable->fetchAll($where);
  99.  
  100.         $grants = array();
  101.         
  102.         foreach($rowSet as $row) {
  103.             $grantedBy        = $row->acl_account;
  104.             $grantedRights    = $row->acl_rights;
  105.             
  106.             // initialize grants to Egwbase_Acl::NONE
  107.             if(!isset($grants[$grantedBy])) {
  108.                 $grants[$grantedBy] = Egwbase_Acl::NONE;
  109.             }
  110.             $grants[$grantedBy] |= $grantedRights;
  111.  
  112.             // if it is a group(negative id) fetch the group members acl too
  113.             if ($grantedBy < 0 && $enumerateGroupAcls === TRUE) {
  114.                 $groupMembers = $accounts->getGroupMembers($grantedBy);
  115.                 
  116.                 foreach($groupMembers as $grantedBy) {
  117.                     // Don't allow to override private with group ACL's!
  118.                     $grantedRights    &= ~Egwbase_Acl::PERSONAL;
  119.                     
  120.                     if(!isset($grants[$grantedBy])) {
  121.                         $grants[$grantedBy] = Egwbase_Acl::NONE;
  122.                     }
  123.                     
  124.                     $grants[$grantedBy] |= $grantedRights;
  125.                 }
  126.             }
  127.         }
  128.         
  129.         // the user has always access to his own data
  130.         $grants[$accountId] = Egwbase_Acl::ANY;
  131.             
  132.         return $grants;
  133.     }

  159.     public function getGrants($accountId, $appName, $requiredRight, $grantType = Egwbase_Acl::ANY_GRANTS)
  160.     {
  161.         $accounts = new Egwbase_Account_Sql();
  162.         $groupMemberships = $accounts->getAccountGroupMemberships($accountId);
  163.         $groupMemberships[] = $accountId;
  164.         
  165.         $aclTable = new Egwbase_Acl_Sql();
  166.         $where = array(
  167.             $aclTable->getAdapter()->quoteInto('acl_appname = ?', $appName),
  168.             // who gave rights to $accountId's groups and $accountId itself
  169.             $aclTable->getAdapter()->quoteInto('acl_location IN (?)', $groupMemberships)
  170.         );
  171.  
  172.         if($grantType === Egwbase_Acl::GROUP_GRANTS) {
  173.             // return groups only (negative id)
  174.             $where[] = 'acl_account < 0';
  175.         } elseif ($grantType === Egwbase_Acl::ACCOUNT_GRANTS) {
  176.             // return accounts only (positive id)
  177.             $where[] = 'acl_account > 0';
  178.         }
  179.         
  180.         $rowSet = $aclTable->fetchAll($where);
  181.  
  182.         $grants = array();
  183.         
  184.         foreach($rowSet as $row) {
  185.             $grantedBy        = $row->acl_account;
  186.             $grantedRights    = $row->acl_rights;
  187.             
  188.             if (!!($grantedRights & $requiredRight)) {
  189.                 // initialize grants to Egwbase_Acl::NONE
  190.                 if(!isset($grants[$grantedBy])) {
  191.                     $grants[$grantedBy] = Egwbase_Acl::NONE;
  192.                 }
  193.                 $grants[$grantedBy] |= $grantedRights;
  194.             }
  195.  
  196.         }
  197.         
  198.         if($grantType == Egwbase_Acl::ANY_GRANTS) {
  199.             // the user has always access to his own data
  200.             $grants[$accountId] = Egwbase_Acl::ANY;
  201.         }
  202.             
  203.         return $grants;
  204.     }
